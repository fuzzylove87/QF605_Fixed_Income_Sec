\begin{flalign*}
V_0 &= D(0,T)g(F) + \int_{0}^{F} h''(K) V^{rec}(K)dK + \int_{F}^{\infty} h''(K) V^{pay}(K)dK
\end{flalign*}

\noindent Where:
\begin{flalign*}
g(K) &= K^{\frac{1}{p}} - 0.04^{\frac{1}{q}}\\
&= K^{\frac{1}{4}}-0.2\\
g'(K) &= \frac{1}{4} K^{-\frac{3}{4}}\\

\end{flalign*}
